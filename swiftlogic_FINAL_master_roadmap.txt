BASE_DEMAND = {
    'silk_kurta':    8,   # units/day at base price, no ads
    'cotton_set':    12,  # highest volume, lowest margin
    'linen_dupatta': 5,   # medium volume, medium margin

def generate_demand(sku, ad_spend, price, competitor_price, base):
    ad_multiplier = 1.0 + (ad_spend / 100.0)  # $100 ads = 2x demand
    price_ratio = competitor_price / max(price, 1.0)
    effective_lambda = base * ad_multiplier * price_ratio
    return int(np.random.poisson(effective_lambda))

def generate_all_demand(state):

Phase 2, Task 3 - Dense Reward Engine
Create reward_model.py. Replace flat reward with dense reward - critical for GRPO training convergence:
def compute_step_reward(action_result, state_before, state_after):
    reward = action_result['base_reward']  # +0.1/+0.3/0.0/-0.2

    # Dense revenue signal every step
    daily_rev = sum(state_after.daily_sales[s]*state_after.prices.get(s,100)
                    for s in state_after.daily_sales)
    reward += 0.001 * daily_rev

    # Solvency bonus
    if state_after.bank_balance >= 500:
        reward += 0.05

    # Stockout penalty
    for sku in state_after.inventory:
        if state_after.inventory[sku]==0 and state_before.inventory.get(sku,0)>0:
            reward -= 0.2

    # Ticket aging penalty (-0.1 per urgent unresolved ticket past day 3)
    urgent = [t for t in state_after.active_tickets
              if t.urgency=='urgent' and t.status=='open'
              and state_after.current_day - t.created_day >= 3]
    reward -= 0.1 * len(urgent)

    # Bankruptcy terminal penalty
    if state_after.bank_balance <= 0:
        reward -= 1.0

    # Reward alignment: training signal aligned with final grader
    reward += 0.01 * (state_after.bank_balance - state_before.bank_balance)

    return round(reward, 4)

def parse_thought_action(text):
    thought_match = re.search(r'Thought:(.*?)Action:', text, re.DOTALL)
    action_match = re.search(r'Action:\s*({.*?})', text, re.DOTALL)
    thought = thought_match.group(1).strip() if thought_match else 'no reasoning'
    try:
        action = json.loads(action_match.group(1)) if action_match else {'action_type':'wait'}
    except:
        action = {'action_type': 'wait'}
    return thought, action

# env/world_engine.py
import json, copy, random
import numpy as np
from pathlib import Path

class WorldEngine:
    def __init__(self, config_path: str = 'configs/siyaani_fashion.json'):
        self.load_config(config_path)

    def load_config(self, config_path: str):
        with open(config_path) as f:
            self.config = json.load(f)
        self._validate_config()
        self._build_lookup_tables()

    def _build_lookup_tables(self):
        self.unit_costs = {p['sku']:p['unit_cost'] for p in self.config['products']}
        self.sell_prices = {p['sku']:p['sell_price'] for p in self.config['products']}
        self.demand_configs = {p['sku']:p['demand'] for p in self.config['products']}

    def reset(self, seed=None):
        if seed: random.seed(seed); np.random.seed(seed)
        cfg = self.config
        self.state = {
            'current_day': 1, 'current_week': 0, 'step_count': 0,
            'bank_balance': cfg['financials']['initial_bank_balance'],
            'inventory': {p['sku']:p['initial_stock'] for p in cfg['products']},
            'prices': self.sell_prices.copy(),
            'active_tickets': self._generate_tickets(),
            'daily_sales': {p['sku']:0 for p in cfg['products']},
            'active_ad_spend': {p['sku']:0.0 for p in cfg['products']},
            'cumulative_revenue': 0.0,
            'done': False,
        }
        return copy.deepcopy(self.state)

    def step(self, action: dict):
        state_before = copy.deepcopy(self.state)
        reward = self._process_action(action)
        self._simulate_day()
        reward = self._compute_reward(reward, state_before)
        self.state['step_count'] += 1
        done = (self.state['step_count'] >= self.config['episode']['max_steps']
                or self.state['bank_balance'] <= self.config['financials']['bankruptcy_threshold'])
        self.state['done'] = done
        return copy.deepcopy(self.state), reward, done

    def _process_action(self, action):
        atype = action.get('action_type')
        if atype not in self.config['actions']['allowed']:
            return self.config['rewards']['invalid_action']
        if atype == 'restock':
            return self._do_restock(action)
        elif atype == 'refund':
            return self._do_refund(action)
        elif atype == 'ad_spend':
            return self._do_ad_spend(action)
        return self.config['rewards']['wait']

    def _simulate_day(self):
        # Poisson demand per product per config
        for product in self.config['products']:
            sku = product['sku']
            dcfg = product['demand']
            ad = self.state['active_ad_spend'].get(sku, 0)
            ad_mult = 1.0 + (ad / 100.0) * dcfg.get('ad_elasticity', 1.0)
            day_of_week = self.state['current_day'] % 7
            season = dcfg.get('seasonality_weights', [1.0]*7)[day_of_week]
            lam = dcfg['base_units_per_day'] * ad_mult * season
            demand = int(np.random.poisson(lam))
            sold = min(demand, self.state['inventory'].get(sku, 0))
            self.state['daily_sales'][sku] = sold
            self.state['inventory'][sku] = max(0, self.state['inventory'].get(sku,0) - sold)
            revenue = sold * self.state['prices'].get(sku, 0)
            self.state['bank_balance'] += revenue
            self.state['cumulative_revenue'] += revenue
        self.state['current_day'] += 1
        self.state['current_week'] = self.state['current_day'] // 7

# supplier_agent.py - rule-based, no second LLM needed
class SupplierAgent:
    BASE_PRICES = {'silk_kurta':800,'cotton_set':500,'linen_dupatta':600,'silk_saree':1200}

    def quote_price(self, sku, quantity_requested, demand_signal):
        base = self.BASE_PRICES.get(sku, 700)
        volume_premium = 1.0 + max(0, (quantity_requested-20)*0.01)
        demand_premium = 1.0 + (demand_signal * 0.1)  # high demand = higher price
        return round(base * volume_premium * demand_premium, 2)
